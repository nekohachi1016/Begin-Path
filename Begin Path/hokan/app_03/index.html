超シンプル 水面の揺らぎ

　

リアルな物理シミュレーションではなく、sin波のゆらぎを横方向に動かすだけの簡易版。
※JavaScriptのコードは直接HTMLファイル内に埋め込んでいます。


▶ 実行してみる 


        
<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>超シンプル 水面の揺らぎ</title>
  <style>
    html, body {
      margin: 0;
      padding: 0;
      background: black;
      overflow: hidden;
    }
    canvas {
      display: block;
    }
  </style>
</head>
<body>
<canvas id="canvas"></canvas>
<script>
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");

canvas.width = window.innerWidth;
canvas.height = window.innerHeight;

let t = 0; // 時間

function animate(){
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  for(let y = 0; y < canvas.height; y += 4){
    ctx.beginPath();
    for(let x = 0; x < canvas.width; x += 4){
      // sin波で揺らぎを作る
      let offset = Math.sin((x + t) * 0.02) * 5;
      ctx.fillStyle = `hsl(${180 + offset*10}, 70%, 60%)`;
      ctx.fillRect(x, y + offset, 4, 4);
    }
  }

  t += 1; // 時間を進める
  requestAnimationFrame(animate);
}

animate();
</script>
</body>
</html>
